[HttpPost]
public async Task<IActionResult> AddReview(CreateReviewDto dto)
{
    int userId = GetUserId();

    // Kiểm tra user có đơn hàng COMPLETED chứa sản phẩm này không
    bool hasPurchased = await _orderRepo
        .HasCompletedOrderWithProductAsync(userId, dto.ProductId);

    if (!hasPurchased)
        return BadRequest("Bạn chưa mua sản phẩm này.");

    // Kiểm tra đã review chưa (unique constraint sẽ bắt ở DB, nhưng check trước cho UX tốt)
    bool alreadyReviewed = await _reviewRepo
        .ExistsAsync(userId, dto.ProductId);

    if (alreadyReviewed)
        return Conflict("Bạn đã đánh giá sản phẩm này rồi.");

    var review = await _reviewRepo.AddAsync(userId, dto);
    return Ok(review);
}

// IHostedService chạy nền, tự động quét mỗi 60 giây
public class StockAlertService : BackgroundService
{
    private const int LOW_STOCK_THRESHOLD = 5; // ngưỡng cảnh báo

    protected override async Task ExecuteAsync(CancellationToken ct)
    {
        while (!ct.IsCancellationRequested)
        {
            // Lấy danh sách sản phẩm có Stock <= ngưỡng
            var lowStockProducts = await _productRepo
                .GetLowStockAsync(LOW_STOCK_THRESHOLD);

            foreach (var product in lowStockProducts)
            {
                // Tránh tạo alert trùng nếu chưa được giải quyết
                bool hasUnread = await _alertRepo
                    .HasUnreadAlertAsync(product.Id);

                if (!hasUnread)
                    await _alertRepo.CreateAsync(product.Id, product.Stock);
            }

            await Task.Delay(TimeSpan.FromSeconds(60), ct);
        }
    }
}
Mục tiêu học: BackgroundService trong .NET — chạy tác vụ định kỳ mà không cần cron job bên ngoài.

[HttpPost("validate")]
[Authorize]
public async Task<IActionResult> Validate(ValidateCouponDto dto)
{
    var coupon = await _couponRepo.FindByCodeAsync(dto.Code);

    if (coupon == null || !coupon.IsActive)
        return NotFound("Mã không hợp lệ.");

    if (coupon.ExpiresAt < DateTime.UtcNow)
        return BadRequest("Mã đã hết hạn.");

    if (coupon.UsedCount >= coupon.MaxUses)
        return BadRequest("Mã đã được dùng đủ số lần cho phép.");

    if (dto.OrderTotal < coupon.MinOrderValue)
        return BadRequest($"Đơn hàng phải từ {coupon.MinOrderValue:N0}đ trở lên.");

    // Trả về số tiền được giảm để hiện preview cho user
    decimal discount = coupon.DiscountType == "percent"
        ? dto.OrderTotal * coupon.DiscountValue / 100
        : coupon.DiscountValue;

    return Ok(new { discount, message = "Áp dụng thành công!" });
}
Mục tiêu học: Viết các guard clause tuần tự để validate business rules — mỗi điều kiện một return rõ ràng, không lồng if.

1

File sửa: BaseCore.APIService/Controllers/ReportsController.cs

[HttpGet("orders/export")]
[Authorize(Roles = "Admin")]
public async Task<IActionResult> ExportOrders(
    [FromQuery] DateTime? from,
    [FromQuery] DateTime? to)
{
    var orders = await _orderRepo.GetByDateRangeAsync(from, to);

    // Dùng StringBuilder để build CSV trong bộ nhớ (không cần thư viện ngoài)
    var sb = new StringBuilder();
    sb.AppendLine("OrderId,CustomerName,Total,Status,CreatedAt");

    foreach (var o in orders)
        sb.AppendLine($"{o.Id},{o.User.Username},{o.Total},{o.Status},{o.CreatedAt:yyyy-MM-dd}");

    // Trả về file download — browser tự mở hộp thoại Save As
    var bytes = Encoding.UTF8.GetBytes(sb.ToString());
    return File(bytes, "text/csv", $"orders-{DateTime.Now:yyyyMMdd}.csv");
}
Mục tiêu học: File() action result trong ASP.NET Core — trả về binary content thay vì JSON.

const handleExport = async () => {

public async Task<IActionResult> Search([FromQuery] string? q)

Autocomplete với Debounce trên React 
// Chỉ gọi API sau 300ms user ngừng gõ — tránh spam request
const [suggestions, setSuggestions] = useState([]);

useEffect(() => {
    if (!query.trim()) return;
    const timer = setTimeout(async () => {
        const res = await api.get(`/api/search/suggestions?q=${query}`);
        setSuggestions(res.data);
    }, 300);                        // debounce 300ms
    return () => clearTimeout(timer); // cleanup khi query thay đổi
}, [query]);
